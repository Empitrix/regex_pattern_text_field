    * github = r'((git|ssh|http(s)?)|(git@[\w\.]+))(:(\/\/)?)([\w\.@\:/\-~]+)(\.git)(\/)?'
    * twitter = r'(?:(?:twitter?|tw): @)+[\w/\-_?=%.]+'
    * facebook = r'(?:(?:facebook?|fb?|Facebook): @)+[\w/\-_?=%.]+'
    * instagram = r'(?:(?:instagram?|insta?|Instagram): @)+[\w/\-_?=%.]+'
    * dateTime = r'\d{1,2} [a-zA-Z]{3} \d{4}'
